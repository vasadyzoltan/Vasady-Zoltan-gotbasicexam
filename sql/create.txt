CREATE TABLE characters(
  id int NOT NULL,
  name varchar(100) NOT NULL,
  alias varchar(100),
  dead varchar(100),
  portrait varchar(100) NOT NULL,
  picture varchar(100) NOT NULL,
  bio varchar(600),
  house varchar(100),
  organization varchar(100),
  PRIMARY KEY (id, portrait, image),

)
  ENGINE=InnoDB;
