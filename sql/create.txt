CREATE TABLE characters(
  Azonosító int NOT NULL,
  Név varchar(100) NOT NULL,
  Becenév varchar(100),
  Halott_e varchar(5),
  Portrékép varchar(100) NOT NULL,
  Kép varchar(100) NOT NULL,
  Történet varchar(600),
  Ház varchar(100),
  Szervezet varchar(100),
  PRIMARY KEY (Azonosító)
)
  ENGINE=InnoDB;
